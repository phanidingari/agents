# Code to calculate the first 10,000 terms of the series 1 - 1/3 + 1/5 - 1/7 + ...

def calculate_series(n):
    total = 0.0
    for i in range(n):
        term = (-1) ** i / (2 * i + 1)  # Calculate each term
        total += term  # Add the term to the total
    return total * 4  # Multiply the result by 4

result = calculate_series(10000)
